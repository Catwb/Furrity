<script>
const timers = [];

// 添加计时器（只需日期，如 "2026-06-07"）
function addTimer(dateStr, title) {
    const id = 'timer-' + Date.now();
    const p = document.createElement('p');
    p.id = id;
    document.getElementById('countdown').appendChild(p); // 或指定容器：document.getElementById('容器id').appendChild(p)
    
    timers.push({
        id: id,
        el: p,
        target: new Date(dateStr + 'T00:00:00').getTime(), // 自动补全0点
        title: title || ''
    });
}

// 更新所有计时器
function update() {
    const now = Date.now();
    
    for (let i = timers.length - 1; i >= 0; i--) {
        const t = timers[i];
        const left = t.target - now;
        
        if (left <= 0) {
            // 时间到，移除
            t.el.textContent = t.title ? t.title + '：已结束' : '倒计时结束';
            setTimeout(() => t.el.remove(), 2000);
            timers.splice(i, 1);
            continue;
        }
        
        const days = Math.floor(left / 86400000);
        const hours = Math.floor((left % 86400000) / 3600000);
        const mins = Math.floor((left % 3600000) / 60000);
        const secs = Math.floor((left % 60000) / 1000);
        
        const timeStr = days + '天' + hours + '时' + mins + '分' + secs + '秒';
        t.el.textContent = t.title ? t.title + '：' + timeStr : timeStr;
    }
}

// 启动
setInterval(update, 1000);
addTimer('2026-06-06', '高考');
</script>
